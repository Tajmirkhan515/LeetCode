class Solution(object):
    def minSubArrayLen(self, target, nums):
        """
        :type target: int
        :type nums: List[int]
        :rtype: int
        """
        left=0
        right=0

        sum=0
        minLength=sys.maxsize 

        for right in range(len(nums)):
            sum+=nums[right]

            while sum>=target:
                minLength=min(minLength,(right-left+1))
                sum-=nums[left]
                left+=1

        return 0 if minLength == sys.maxsize else minLength
