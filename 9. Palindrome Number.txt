class Solution(object):
    def isPalindrome(self, x):

        if x<0:
            return False
        arr=[]
        num=x
        while num>0:
            arr.append(num%10)
            num=num//10
        
        print(arr)
        
        l=0
        r=len(arr)-1

        while l<=r:
            if arr[l]!=arr[r]:
                return False
            l+=1
            r-=1
        return True
