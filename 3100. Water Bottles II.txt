class Solution(object):
    def maxBottlesDrunk(self, numBottles, numExchange):
        """
        :type numBottles: int
        :type numExchange: int
        :rtype: int
        """
        empty=numBottles
        drunk=numBottles

        while empty>=numExchange:
            empty=(empty-numExchange)+1
            numExchange+=1
            drunk+=1
        
        return drunk
