# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution(object):
    def middleNode(self, head):
        
        if head==None or head.next==None:
            return head
        
        fast=head
        slow=head

        while fast!=None and fast.next!=None:
            slow=slow.next
            fast=fast.next.next
        
        return slow
