class Solution(object):
    def isValid(self, s):
        
        
        stk=[]

        for i in range(0,len(s)):
            if s[i]=='(' or s[i]=='[' or s[i]=='{':
                stk.append(s[i])
            
            else:
                if not stk:
                    return False
                
                if stk[-1]=='(' and s[i]==')':
                    stk.pop()
                elif stk[-1]=='{' and s[i]=='}':
                    stk.pop()
                elif stk[-1]=='[' and s[i]==']':
                    stk.pop()
                
                else:
                    return False
        return not stk
