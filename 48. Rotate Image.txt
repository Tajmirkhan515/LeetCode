class Solution(object):
    def rotate(self, matrix):
        
        lengthOfM=len(matrix)

        for i in range(0, lengthOfM):
            for j in range (i+1, lengthOfM):
                matrix[i][j], matrix[j][i]=matrix[j][i], matrix[i][j]

        for i in range(0,lengthOfM):
            matrix[i].reverse()

        return matrix
