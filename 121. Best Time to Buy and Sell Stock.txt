class Solution(object):
    def maxProfit(self, prices):
        max=0
        actualprice=0
        for i in range (len (prices)):
            actualprice=prices[i]
            localmax=0
            for j in range(i+1,len(prices)):
                if (prices[j]-actualprice)>localmax:
                    localmax=prices[j]-actualprice
                
            if localmax>max:
                max=localmax
        return max
