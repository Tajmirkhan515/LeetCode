class Solution(object):
    def mySqrt(self, x):
        l=0
        r=x
        result=0

        while l<=r:
            mid=(l+r)//2

            if mid*mid==x:
                return mid

            elif mid*mid<x:
                l=mid+1
                result=mid

            elif mid*mid>x:
                r=mid-1     

        return result
