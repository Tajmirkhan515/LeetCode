class Solution(object):
    def numIslands(self, grid):
        """
        :type grid: List[List[str]]
        :rtype: int
        """
        rows=len(grid)
        cols=len(grid[0])

        direction=[(1,0),(-1,0),(0,-1),(0,1)]
        island=0

        def dfs(r,c):
            if r<0 or c<0 or r>=rows or c>=cols or grid[r][c]=="0":
                return
                
            grid[r][c]="0"
            dfs(r-1,c)
            dfs(r+1,c)
            dfs(r,c+1)
            dfs(r,c-1)

        
        for r in range(rows):
            for c in range(cols):
                if grid[r][c]=="1":
                    island+=1
                    dfs(r,c)
        
        return island
